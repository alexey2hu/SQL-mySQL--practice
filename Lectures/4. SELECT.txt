-- Лучше явно указывать столбцы для получения, во избежание нагрузки на сеть

/*
** Можно проводить логические операции при получении данных
** Доступные операции ('+', '-', '*', '/', '%')
** 
** '%' - модуль (Остаток от деления)
*/

/*
** 'AS' - Присвоение псевдонима (название колонки)
** Для использования пробелов заключить в кавычки ('' или "")
*/
	# Пример:
		SELECT first_name,
			last_name,
			points,
			(points + 10) * 100 AS 'discount factor'
		FROM customers;

/*
** 'DISTINCT' - Возврат только уникальных записей (без дубликатов)
** Используется совместно с инструкцией SELECT
*/

	# Пример:
		SELECT DISTINCT state
		FROM customers;
